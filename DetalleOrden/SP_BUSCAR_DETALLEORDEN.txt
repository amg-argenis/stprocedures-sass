DELIMITER $$
CREATE OR REPLACE PROCEDURE SP_BUSCAR_DETALLEORDEN(
    IN  pa_tenantid       VARCHAR(255),
    IN  pa_iddetalleorden VARCHAR(255),
    IN  pa_ordenid        VARCHAR(255),
    OUT pa_codigobd       INT,
    OUT pa_mensaje        VARCHAR(255)
)
BEGIN
    DECLARE v_sqlstate      CHAR(5);
    DECLARE v_error_message TEXT;
    DECLARE v_count         INT DEFAULT 0;

    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        GET DIAGNOSTICS CONDITION 1
            v_sqlstate      = RETURNED_SQLSTATE,
            v_error_message = MESSAGE_TEXT;
        SET pa_codigobd = -1;
        SET pa_mensaje  = CONCAT('Error desde MySQL: ', v_sqlstate, ': ', v_error_message);
    END;

    SELECT COUNT(idDetalleOrden)
    INTO v_count
    FROM tadetalleordenservicio
    WHERE idDetalleOrden = pa_iddetalleorden
      AND ordenId        = pa_ordenid
      AND tenantId       = pa_tenantid;

    IF v_count = 0 THEN
        SET pa_codigobd = 1;
        SET pa_mensaje  = 'Orden detalle de servicio no encontrada desde MySQL';
    ELSE
        SET pa_codigobd = 0;
        SET pa_mensaje  = 'Orden detalle de servicio encontrada desde MySQL';

        SELECT
            idDetalleOrden,
            ordenId,
            procesoId,
            tipoPrenda,
            cantidad,
            colorReferencia,
            tenantId
        FROM tadetalleordenservicio
        WHERE idDetalleOrden = pa_iddetalleorden
          AND ordenId        = pa_ordenid
          AND tenantId       = pa_tenantid;

    END IF;

END$$
DELIMITER ;
