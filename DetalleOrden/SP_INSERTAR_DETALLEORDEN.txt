DELIMITER $$
CREATE OR REPLACE PROCEDURE SP_INSERTAR_DETALLEORDEN(
    -- IN
    IN  pa_idordendetalle VARCHAR(255),
    IN  pa_ordenid VARCHAR(255),
    IN  pa_procesoid VARCHAR(255),
    IN  pa_tipoprenda VARCHAR(100),
    IN  pa_cantidad INT,
    IN  pa_colorreferencia VARCHAR(255),
    IN  pa_tenantid VARCHAR(255),
  
    -- OUT
    OUT pa_codigobd INT,
    OUT pa_mensaje VARCHAR(255)
)
BEGIN
    DECLARE v_sqlstate CHAR(5);
    DECLARE v_error_message TEXT;
    
    -- Handler de error y rollback
    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        ROLLBACK;
        GET DIAGNOSTICS CONDITION 1
            v_sqlstate = RETURNED_SQLSTATE,
            v_error_message = MESSAGE_TEXT;
        SET pa_codigobd = 1;
        SET pa_mensaje = CONCAT(
            'Error SQLSTATE orden servicio: ',
            v_sqlstate, ' - ', v_error_message
        );
    END;
    
    -- Iniciar transaccion
    START TRANSACTION;
    
    INSERT INTO taordenservicio (
        idDetalleOrden, 
        ordenId, 
        procesoId, 
        tipoPrenda, 
        cantidad, 
        colorReferencia, 
        tenantId
    )
    VALUES (
        pa_idordendetalle,
        pa_ordenid,
        pa_procesoid,
        pa_tipoprenda,
        pa_cantidad,
        pa_colorreferencia,
        pa_tenantid
    );
    
    -- Validacion: asegurar que se inserto
    IF ROW_COUNT() = 0 THEN
        SIGNAL SQLSTATE '45000'
            SET MESSAGE_TEXT = 'No se pudo insertar la orden detalle desde MySQL';
    END IF;
    
    -- Confirmar cambios
    COMMIT;
    
    SET pa_codigobd = 0;
    SET pa_mensaje = 'Orden detalle insertada correctamente desde MySQL';
END$$
DELIMITER ;
