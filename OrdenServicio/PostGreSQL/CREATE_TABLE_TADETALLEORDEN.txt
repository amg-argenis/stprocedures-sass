CREATE TABLE tadetalleorden (
    idDetalleOrden  CHAR(36)      NOT NULL,
    ordenId         CHAR(36),
    procesoId       CHAR(36),
    tipoPrenda      VARCHAR(100),
    cantidad        INTEGER       NOT NULL,
    colorReferencia VARCHAR(100),
    tenantId        CHAR(36),

    CONSTRAINT pk_tadetalleorden PRIMARY KEY (idDetalleOrden),
    CONSTRAINT fk_tadetalleorden_orden 
        FOREIGN KEY (ordenId) REFERENCES taordenservicio(idOrden)
);

select * from tadetalleorden;

CREATE INDEX idx_tadetalleorden_idDetalleOrden  ON tadetalleorden(idDetalleOrden);
CREATE INDEX idx_tadetalleorden_ordenId   ON tadetalleorden(ordenId);
CREATE INDEX idx_tadetalleorden_procesoId ON tadetalleorden(procesoId);
CREATE INDEX idx_tadetalleorden_tenantId  ON tadetalleorden(tenantId);
