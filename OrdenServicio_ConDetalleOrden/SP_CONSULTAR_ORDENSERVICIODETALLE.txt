DELIMITER $$

CREATE PROCEDURE SP_BUSCAR_ORDENSERVICIOCONDETALLE (
    IN  pa_idorden   CHAR(36),
    IN  pa_folio     VARCHAR(50),

    OUT pa_codigobd  INT,
    OUT pa_mensajebd VARCHAR(255)
)
BEGIN
    DECLARE vl_existe INT DEFAULT 0;

    DECLARE EXIT HANDLER FOR SQLEXCEPTION
    BEGIN
        SET pa_codigobd  = 1;
        SET pa_mensajebd = 'Error interno al consultar la orden de servicio con detalle, desde MySQL';
    END;

    SELECT COUNT(idOrden)
    INTO   vl_existe
    FROM   taordenservicio
    WHERE  idOrden = pa_idorden
    AND    folio   = pa_folio;

    IF vl_existe = 0 THEN
        SET pa_codigobd  = 2;
        SET pa_mensajebd = 'Orden de servicio con detalle no encontrada, desde MySQL';
    ELSE
        -- Un solo result set: N filas (una por cada detalle)
        SELECT
            o.idOrden,
            o.clienteId,
            o.folio,
            o.fechaIngreso,
            o.estado,
            o.totalPrendas,
            o.observaciones,
            o.createdAt,
            o.tenantId,
            o.fechaEntrega,
            d.idDetalleOrden,
            d.ordenId,
            d.procesoId,
            d.tipoPrenda,
            d.cantidad,
            d.colorReferencia,
            d.tenantId AS tenantIdDetalle
        FROM  taordenservicio o
        INNER JOIN tadetalleorden d
            ON  d.ordenId  = o.idOrden
            AND d.tenantId = o.tenantId
        WHERE o.idOrden = pa_idorden
        AND   o.folio   = pa_folio;

        SET pa_codigobd  = 0;
        SET pa_mensajebd = 'Consulta exitosa de orden servicio con detalle, desde MySQL';
    END IF;

END$$

DELIMITER ;
